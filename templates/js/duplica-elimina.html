// ===== DUPLICA / ELIMINA ANAGRAFICA ===== window.apriDuplicaAnagrafica = function(tipo, pk) { var nomi = { 'datore': 'Datore', 'lavoratore': 'Lavoratore', 'beneficiario': 'Beneficiario' }; loadAjaxForm('/ajax/duplica-anagrafica/' + tipo + '/' + pk + '/', 'Duplica ' + (nomi[tipo] || tipo)); }; window.confermaEliminaAnagrafica = function(tipo, pk) { if (!confirm('Eliminare definitivamente questa anagrafica? Operazione irreversibile.')) return; var fd = new FormData(); fd.append('csrfmiddlewaretoken', document.querySelector('[name=csrfmiddlewaretoken]').value); fetch('/ajax/elimina-anagrafica/' + tipo + '/' + pk + '/', { method: 'POST', body: fd, headers: { 'X-Requested-With': 'XMLHttpRequest' } }) .then(function(r) { return r.json(); }) .then(function(d) { if (d.success) { var offcanvas = bootstrap.Offcanvas.getInstance(document.getElementById('panelNuovo')); if (offcanvas) offcanvas.hide(); var tbody = document.querySelector('table.custom-table tbody'); mostraToast('Anagrafica eliminata', true); if (tbody) setTimeout(function() { window.location.reload(); }, 1000); } else { mostraToast(d.errore || 'Errore durante l\'eliminazione', false); } }).catch(function(e) { mostraToast('Errore: ' + e.message, false); }); }; window.apriDuplicaTabella = function(tipo, pk) { loadAjaxForm('/ajax/duplica-tabella/' + tipo + '/' + pk + '/', 'Duplica'); }; window.confermaEliminaTabella = function(tipo, pk) { if (!confirm('Eliminare definitivamente questo elemento? Operazione irreversibile.')) return; var fd = new FormData(); fd.append('csrfmiddlewaretoken', document.querySelector('[name=csrfmiddlewaretoken]').value); fetch('/ajax/elimina-tabella/' + tipo + '/' + pk + '/', { method: 'POST', body: fd, headers: { 'X-Requested-With': 'XMLHttpRequest' } }) .then(function(r) { return r.json(); }) .then(function(d) { if (d.success) { var offcanvas = bootstrap.Offcanvas.getInstance(document.getElementById('panelNuovo')); if (offcanvas) offcanvas.hide(); var tbody = document.querySelector('table.custom-table tbody'); mostraToast('Elemento eliminato', true); if (tbody) setTimeout(function() { window.location.reload(); }, 1000); } else { mostraToast(d.errore || 'Errore durante l\'eliminazione', false); } }).catch(function(e) { mostraToast('Errore: ' + e.message, false); }); };
