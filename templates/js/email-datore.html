window.anteprimaEmailDatore = function(pk) { var sel = document.getElementById('modelloMail'); if (!sel) return; var modelloPk = sel.value; if (!modelloPk) { mostraToast('Seleziona un modello email.', false); return; } var oldBox = document.getElementById('emailPreviewDatoreFrame'); if (oldBox) { oldBox.parentElement.removeChild(oldBox); } fetch('/ajax/anteprima-email-datore/' + pk + '/', { method: 'POST', headers: {'Content-Type':'application/json','X-CSRFToken': getCSRFToken()}, body: JSON.stringify({modello_mail_pk: modelloPk}) }) .then(function(r) { return r.json(); }) .then(function(data) { if (data.error) { mostraToast(data.error, false); return; } var box = document.createElement('div'); box.id = 'emailPreviewDatoreFrame'; box.style.cssText = 'position:fixed;top:80px;right:24px;width:560px;background:#17181C;border:1px solid #27282D;border-radius:12px;box-shadow:0 20px 60px rgba(0,0,0,0.5);z-index:10050;font-family:inherit;'; box.innerHTML = '
' + 'Anteprima Email Datore' + '' + '
' + '
' + (data.oggetto || '') + '
' + '
' + (data.corpo || '') + '
'; document.body.appendChild(box); }) .catch(function() { mostraToast('Errore caricamento anteprima.', false); }); };
