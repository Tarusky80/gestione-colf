function toggleAnticipiTFR(el, pk) { var section = document.getElementById('anticipiTfrSection'); if (!section) return; var vis = section.style.display !== 'none'; section.style.display = vis ? 'none' : 'block'; var chevron = el.querySelector('.bi-chevron-down'); if (chevron) chevron.style.transform = vis ? '' : 'rotate(180deg)'; if (!vis) caricaAnticipiTFR(pk); } function caricaAnticipiTFR(pk) { if (!pk) { var inp = document.getElementById('dynamicForm'); if (inp) pk = inp.getAttribute('data-contratto-pk'); } if (!pk) { var m = document.querySelector('[data-contratto-pk]'); if (m) pk = m.getAttribute('data-contratto-pk'); } if (!pk) return; fetch('/ajax/lista-anticipi-tfr/' + pk + '/') .then(function(r) { return r.json(); }) .then(function(d) { if (!d.success) return; var list = document.getElementById('anticipiTfrList'); if (!list) return; var html = ''; if (d.items.length === 0) { html = '
Nessun anticipo TFR registrato.
'; } else { html = '
Storico anticipi
'; for (var i = 0; i < d.items.length; i++) { var a = d.items[i]; html += '
'; html += '\u20ac ' + a.importo + ''; html += '' + a.data + ''; if (a.note) html += '(' + a.note + ')'; html += ''; html += '
'; } html += '
Totale: \u20ac ' + d.totale + '
'; } list.innerHTML = html; var cnt = document.getElementById('anticipiTfrCount'); if (cnt) cnt.textContent = '(' + d.items.length + ')'; }); } function salvaAnticipoTFR(contrattoPk) { var importo = document.getElementById('anticipoImporto'); var data = document.getElementById('anticipoData'); var note = document.getElementById('anticipoNote'); if (!importo || !importo.value) { mostraToast('Inserire un importo.', false); return; } var csrfSalva = document.querySelector('[name=csrfmiddlewaretoken]'); var tokenSalva = csrfSalva ? csrfSalva.value : ''; fetch('/ajax/salva-anticipo-tfr/', { method: 'POST', headers: { 'X-CSRFToken': tokenSalva, 'Content-Type': 'application/json' }, body: JSON.stringify({ contratto_pk: contrattoPk, importo: importo.value, data: data ? data.value : '', note: note ? note.value : '' }) }) .then(function(r) { return r.json(); }) .then(function(d) { if (d.success) { mostraToast('Anticipo TFR salvato.', true); if (importo) importo.value = ''; if (note) note.value = ''; if (data) data.value = ''; caricaAnticipiTFR(contrattoPk); } else { mostraToast('Errore: ' + (d.error || 'Salvataggio fallito.'), false); } }) .catch(function(err) { mostraToast('Errore di rete: ' + err.message, false); }); } function eliminaAnticipoTFR(pk) { if (!confirm('Eliminare questo anticipo TFR?')) return; var csrf = document.querySelector('[name=csrfmiddlewaretoken]'); var token = csrf ? csrf.value : ''; fetch('/ajax/elimina-anticipo-tfr/' + pk + '/', { method: 'POST', headers: { 'X-CSRFToken': token } }) .then(function(r) { var ct = r.headers.get('content-type') || ''; if (ct.indexOf('application/json') === -1 && ct.indexOf('text/json') === -1) { throw new Error('Risposta non valida dal server'); } return r.json(); }) .then(function(d) { if (d.success) { mostraToast('Anticipo eliminato.', true); caricaAnticipiTFR(null); } else { mostraToast('Errore: ' + (d.error || 'Eliminazione fallita.'), false); } }) .catch(function(err) { mostraToast('Errore di rete: ' + err.message, false); }); }
